find(condition) → {Array}
hashMap.js, line 325

조건을 체크하는 콜백을 전달받아 데이터를 전달해주고 콜백의 결과가 true인경우의 데이터를 모와 배열로 만들어 리턴해준다.

Name Type Description
condition function
Returns:
Array
Example
//ex1
var hm = new HashMap();
hm.set('key', 'value');
hm.set('key2', 'value');

hm.find(function(value, key) {
    return key === 'key2';
}); // ['value']

//ex2
var hm = new HashMap({
    'myobj1': {
         visible: true
     },
    'mybobj2': {
         visible: false
     }
});

hm.find(function(obj, key) {
    return obj.visible === true;
}); // [{visible: true}];
