find(condition) → {Array}
hashMap.js, line 327

Work similarly to Array.prototype.map().
It executes the provided callback that checks conditions once for each element of hashMap,
and returns a new array having elements satisfying the conditions

Name Type Description
condition function

A function that checks conditions
Returns:

A new array having elements satisfying the conditions

Example
//ex1
 var hm = new HashMap();
 hm.set('key', 'value');
 hm.set('key2', 'value');

 hm.find(function(value, key) {
     return key === 'key2';
 }); // ['value']

 //ex2
 var hm = new HashMap({
     'myobj1': {
          visible: true
      },
     'mybobj2': {
          visible: false
      }
 });

 hm.find(function(obj, key) {
     return obj.visible === true;
 }); // [{visible: true}];
