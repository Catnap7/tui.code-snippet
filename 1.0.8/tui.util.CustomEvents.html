customEvent.off('onload');

// 2. off by event name and handler

// 3. off by handler
customEvent.off(handler);

// 4. off by context
customEvent.off(myObj);

// 5. off by context and handler
customEvent.off(myObj, handler);

// 6. off by context and event name
customEvent.off(myObj, 'onload');

// 7. off by an Object.<string, function> that is {eventName: handler}
customEvent.off({
  'play': handler,
  'pause': handler2
});

// 8. off the all events
customEvent.off();

invoke(eventName, …data) → {boolean}
customEvent.js, line 522

Fire a event and returns the result of operation 'boolean AND' with all listener's results.

So, It is different from CustomEvents#fire.

In service code, use this as a before event in component level usually for notifying that the event is cancelable.

Name Type Attributes Description
eventName string

Custom event name
data * <repeatable>

Data for event
Returns:

The result of operation 'boolean AND'

Example
if (this.invoke('beforeZoom')) {    // check the result of 'beforeZoom'
     // if true,
     // doSomething
 }

 // In service code,
 map.on({
     'beforeZoom': function() {
         // It should cancel the 'zoom' event by some conditions.
         if (that.disabled && this.getState()) {
             return false;
         }
         return true;
     }
 });
